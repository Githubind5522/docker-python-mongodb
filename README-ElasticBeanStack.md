# Elastic Bean Stack

# Pre-Requisites
    Install python
    Install pip
    Install Virtualenv
# Install python and pip
    yum install python-pip -y
# Install Virtualenv
    pip install virtualenv
# Create virtual environment for application
    virtualenv helloworld
# Zip all files of application
